    <script>
        // Initial Page Entrance
        window.addEventListener('load', () => {
            gsap.to("#main-card", {
                duration: 1.5,
                opacity: 1,
                y: -20,
                ease: "power4.out"
            });
            gsap.from(".reveal-text", {
                duration: 1,
                opacity: 0,
                x: -20,
                stagger: 0.3,
                delay: 1
            });
        });

        // Function to create floating hearts
        function createHeart() {
            const container = document.getElementById('heart-container');
            const heart = document.createElement('i');
            heart.classList.add('fa-solid', 'fa-heart', 'heart');
            
            // Random position and size
            const size = Math.random() * 20 + 10 + "px";
            heart.style.left = Math.random() * 100 + "vw";
            heart.style.fontSize = size;
            heart.style.opacity = Math.random();
            
            container.appendChild(heart);

            // Animate up
            gsap.to(heart, {
                y: -window.innerHeight - 100,
                x: (Math.random() - 0.5) * 200,
                duration: Math.random() * 3 + 2,
                ease: "none",
                onComplete: () => heart.remove()
            });
        }

        // Triggered by button
        function startMagic() {
            // Intense heart burst
            for(let i=0; i<30; i++) {
                setTimeout(createHeart, i * 100);
            }
            
            // Shake the card
            gsap.to("#main-card", { duration: 0.1, x: 5, repeat: 5, yoyo: true });
            
            // Change button text
            const btn = document.getElementById('magicBtn');
            btn.innerHTML = "You're Special! ðŸ’–";
            btn.style.backgroundColor = "#db2777";

            // Start music if not playing
            const music = document.getElementById('bdayMusic');
            if(music.paused) toggleMusic();
        }

        // Background Hearts Interval
        setInterval(createHeart, 400);

        // Music Toggle Logic
        function toggleMusic() {
            const music = document.getElementById('bdayMusic');
            const icon = document.getElementById('musicIcon');
            if (music.paused) {
                music.play();
                icon.classList.remove('fa-play');
                icon.classList.add('fa-pause');
            } else {
                music.pause();
                icon.classList.remove('fa-pause');
                icon.classList.add('fa-play');
            }
        }
    </script>
